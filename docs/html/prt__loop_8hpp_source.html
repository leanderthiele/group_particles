136 Workspace<AFields>::prt_loop_sorted (size_t Nprt_this_file)
137 {// {{{
138  // create a Sorting instance, constructing it will perform the main work
139  // associated with this object
140  #ifndef NDEBUG
141  TIME_PT(t1);
142  #endif // NDEBUG
143 
144  Sorting prt_sort (Nprt_this_file, Bsize, tmp_prt_properties);
145 
146  #ifndef NDEBUG
147  TIME_MSG(t1, "initialization of Sorting instance (Nprt=%lu)", Nprt_this_file);
148  #endif // NDEBUG
149 
150  #pragma omp parallel
151  {
152 
153  // loop over groups
154  #pragma omp for schedule(dynamic,1)
155  for (size_t grp_idx=0; grp_idx != Ngrp; ++grp_idx)
156  {
157  // we have to initialize this on each iteration,
158  // otherwise the loop doesn't work with OpenMP
159  typename Callback<AFields>::GrpProperties grp (grp_properties, grp_idx);
160 
161  // compute which cells have intersection with this group
162  const std::vector<std::tuple<size_t,size_t,std::array<int,3>>> prt_idx_ranges
163  = prt_sort.prt_idx_ranges(grp.coord(), grp_radii[grp_idx], grp_radii_sq[grp_idx]);
164 
165  // no particles in the vicinity of this group
166  if (prt_idx_ranges.empty()) continue;
167 
168  // loop over cells
169  for (auto &prt_idx_range : prt_idx_ranges)
170  {
171  typename Callback<AFields>::PrtProperties prt (prt_sort.tmp_prt_properties_sorted,
172  std::get<0>(prt_idx_range));
173 
174  // loop over particles
175  for (size_t prt_idx=std::get<0>(prt_idx_range);
176  prt_idx != std::get<1>(prt_idx_range);
177  ++prt_idx, prt.advance())
178  prt_loop_inner(grp_idx, grp, prt, std::get<2>(prt_idx_range));
179  }// for prt_idx_range
180  }// for grp_idx
181 
182  } // parallel
183 }// }}}

194 Workspace<AFields>::prt_loop_inner
195  (size_t grp_idx,
196  const typename Callback<AFields>::GrpProperties &grp,
197  const typename Callback<AFields>::PrtProperties &prt,
198  const std::array<int, 3> &periodic_to_add)
199 #endif // NAIVE
200 {// {{{
201  coord_t *rgrp = grp.coord();
202  coord_t *rprt = prt.coord();
203 
204  #ifdef EARLY_RETURN
205  coord_t Rsq = (coord_t)0.0;
206  for (size_t ii=0; ii != 3; ++ii)
207  {
208  #ifdef NAIVE
209  coord_t dx = GeomUtils::abs_periodic_dist(rgrp[ii], rprt[ii], Bsize);
210  #else // NAIVE
211  coord_t dx = std::fabs(GeomUtils::periodic_dist_whint(rgrp[ii], rprt[ii], Bsize, periodic_to_add[ii]));
212  #endif // NAIVE
213 
214  if (dx > grp_radii[grp_idx])
215  return;
216 
217  Rsq += dx * dx;
218  }
219  #else // EARLY_RETURN
220  #ifndef NAIVE
221  coord_t Rsq = GeomUtils::periodic_hypotsq(rgrp, rprt, Bsize, periodic_to_add);
222  #else // NAIVE
223  coord_t Rsq = GeomUtils::periodic_hypotsq(rgrp, rprt, Bsize);
224  #endif // NAIVE
225  #endif // EARLY_RETURN
226 
227  // check if this particle belongs to the group
228  if (Rsq > grp_radii_sq[grp_idx])
229  return;
230 
231  // particle belongs to group: do the user-defined thing with it
232  callback.prt_action(grp_idx, grp, prt, Rsq);
233 }// }}}
